122  float penalty; // Penalty applied to climbing/going down

125  } else {
126  const float threshold = -len*.5f;
127  if (offset.z() > threshold) penalty = offset.z();
128  else penalty = threshold;
129  }
130  float res = len + penalty;
131  //osg::Vec3 flat(offset.x(), offset.y(), 0); flat.normalize();

133  //TODO tan()? threshold ? float penalty = (offset.z()>0 ? cosAngle : -cosAngle)*.5f + .5f;
134  //if (penalty<.5f) penalty = .5f; // Clamp
135  //float res = len * penalty;
136 
137  ASSERT(res>0); // Else you should double check the penalty
138  return res;

678  //posX = x;
679  //posY = y;
680  windowX = width;
681  windowY = height;
682  ASSERT(windowX>0 && windowY>0);
683 
684  float aspectRatio = windowX/static_cast<float>(windowY);
685  const float refAspectRatio = 4/3.f; // Game is intended to have a 4/3 aspect ratio by default
686  osg::Camera * pCamera = pimpl->viewer.getCamera();
687  // Disable auto ZNear/ZFar (Faster, and so skybox doesn't need to modify CullSettngs on cull traversal, and we do NOT need auto compute !)
688  pCamera->setComputeNearFarMode(osg::CullSettings::DO_NOT_COMPUTE_NEAR_FAR);
689  double fovy=90., basePixelRatio=1., zNear=.05, zFar=10000.;
690  double pixelRatio = basePixelRatio;
691  if (aspectRatio < refAspectRatio || options[OPTION_NAME[OPT_WIDE_SHRINK]].as<bool>()) {
692  // Aspect ratio is < 4/3 or there is an option to horizontally shrink the image for >4/3 resolutions.
693  // Note: We also could move the camera backwards instead of horizontally shrinking the output.
694  pixelRatio = basePixelRatio;
695  } else {
696  // Image is not horizontally shrinked. Resolutions >4/3 will then see more of the world on the left and right sides.
697  pixelRatio = basePixelRatio * (aspectRatio / refAspectRatio);
698  }
699  pCamera->setProjectionMatrixAsPerspective(fovy, pixelRatio, zNear, zFar);
700  pCamera->setProjectionResizePolicy(osg::Camera::FIXED);

704  if (pHud.valid()) pHud->onProjectionChanged(0, width, 0, height);
705  //else remove the pointer from the list
706  }
