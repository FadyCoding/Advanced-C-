178 bool App::readOptions(int argc, char **argv) {
179  namespace po = boost::program_options;
180  const boost::filesystem::path optionsPath(AppPaths::profiles() / "Options.txt");
181  if (!boost::filesystem::exists(optionsPath)) writeOptions(false);
182  ASSERT(boost::filesystem::exists(optionsPath));
183 
184  po::store(po::parse_command_line(argc, argv, cmdOptions), options);
185  std::ifstream ifs(optionsPath.string().c_str());
186  po::store(po::parse_config_file(ifs, fileOptions), options);
187  ifs.close();
188  po::notify(options);
189 
190  return options.count("help")==0;
191 }
