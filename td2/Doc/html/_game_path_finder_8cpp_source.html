19 void PathFinderCommandSimplePath::onPathFinderReady(PathFinder & pf) {
20  pf.compute(start, end);
21 
22  boost::lock_guard<boost::mutex> lock(resultMutex);
23  done = true;
24 
25  // Copy result
26  const auto & p = pf.getPredecessors();
27  auto iEnd = pf.getEndIndex();
28  if (p.empty() || p[iEnd] == iEnd) {
29  // No path found
30  result.clear();
31  return;
32  }
33 
34  // Count
35  auto iStart = pf.getStartIndex();
36  unsigned int nbPoints = 0;
37  for (auto iCur = iEnd; iCur != iStart; iCur = p[iCur], ++nbPoints) {
38  ASSERT(iCur != p[iCur] || iCur == iStart);
39  }
40  ++nbPoints; // Adding the "start" point
41 
42  // Allocate and fill points backwards
43  result.resize(nbPoints);
44  auto iCurPoint = nbPoints-1;
45  for (auto iCur = iEnd; iCur != iStart; iCur = p[iCur], --iCurPoint) {
46  ASSERT(iCurPoint>0 || p[iCur] == iStart);
47  result[iCurPoint] = pf.getCoordFromDescriptor(iCur);
48  }
49  result[0] = pf.getCoordFromDescriptor(iStart);
50 }

133 void GamePathFinder::triggerGraphModified() {
134  // A callback calling this (in operator()) may destroy itself and/or induce looping this function.
135  // This is not the case now, but code should handle that (tricky) case.
136  // TODO handle self-destruction of a command
137 
138  // If looping, exit (after incrementing the counter)
139  if (++triggeringCallbacksRemaining >1) return;
140  // Loop until there are no "triggers" remaining
141  for(;;) {
142  // Cancel orders
143  {
144  boost::lock_guard<boost::recursive_mutex> lock(queueMutex);
145  queue.erase(std::remove_if(queue.begin(), queue.end(), [this](std::shared_ptr<PathFinderCommand> & c) {
146  return c->onPathFinderModified(this);
147  }), queue.end());
148  }
149  // Trigger callbacks
150  if (PVLEGameHolder::instance().get<Game>()->getState() != Game::STATE_STOP) { // Do not trigger callbacks when game is stopped
151  boost::lock_guard<boost::mutex> lock(cbMutex);
152  for(auto & cb : callbacks) cb->operator()(*this);
153  }
154  ASSERT(triggeringCallbacksRemaining>0);
155  if (--triggeringCallbacksRemaining == 0) break; // Condition not in the for() loop to avoid testing it at first iteration
156  LOG_DEBUG_INFO << BOOST_CURRENT_FUNCTION << " looping" << std::endl;
157  }
158 }
