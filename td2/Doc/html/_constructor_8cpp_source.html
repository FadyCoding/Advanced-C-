197 void Constructor::step(dReal elapsed) {
198  UnpilotedUnit::step(elapsed); // Should be called at the end, but there are plenty of returns, and BOOST_SCOPE_EXIT can't work with parent call.
199  if (testHasDied()) return;
200 
201  // Handle user continuous inputs
202  //ASSERT(pControler);
203  //const ControlState & cs = pControler->getControlState();
204 
205  // First process buildings
206  if (isBuilding()) {
207  // currentlyBuilding may be nullptr if building has been destroyed
208  std::shared_ptr<UnpilotedUnit> currentlyBuildingRef = currentlyBuilding.lock();
209  if (currentlyBuildingRef) {
210  if (!currentlyBuildingRef->isBuilt()) return;
211  }
212  // Done building
213  currentlyBuilding.reset();
214  state = STOPPED;
215  }
216 
217  // Follow path
218  if (state == STOPPED && !pfCommandLaunched) return;
219  boost::lock_guard<boost::mutex> lock(pfMutex);
220 
221  static const float BASE_SPEED = 100;
222  // Gets the result if needed
223  if (pfCommandLaunched && pfCommand->isDone()) {
224  if (pfCommand->getResult().empty()) {
225  LOG_NOTICE << "Constructor could not find a path to destination" << std::endl;
226  if (state == STOPPING) { state = STOPPED; return; }
227  if (state == STOPPED) return;
228  }
229  else {
230  // New path given
231  pathPoints.swap( pfCommand->getResult() );
232  state = MOVING;
233 
234  // TODO remove points at the begining of the path a point is closer (in terms of cost) than the 1st one.
235  // This may happen if the path finder thread took a long time to answer, or if "nearest" point is to the opposite direction.
236 
237  removeDrawnPath();
238  drawnPath = new osg::MatrixTransform();
239  //drawnPath->setReferenceFrame(osg::Transform::ABSOLUTE_RF);
240  drawnPath->addChild(drawPath(pathPoints));
241  pModel->addChild(drawnPath);
242 
243  curPointInPath = 0;
244  pointProgress = 0;
245  pfCommandLaunched = false;
246  zeroPoint = getPos();
247  }
248  }
249 
250  if (state == STOPPED) return;
251 
252  //ASSERT(!pathPoints.empty());
253  if (pathPoints.empty() || curPointInPath >= pathPoints.size()) {
254  cancelMove();
255  return;
256  }
257 
258  osg::Vec3 pos;
259  for(; elapsed>0; ) {
260  const osg::Vec3 & prevPoint = curPointInPath==0 ? zeroPoint : pathPoints[curPointInPath-1];
261  const osg::Vec3 & curPoint = pathPoints[curPointInPath];
262  osg::Vec3 offset(curPoint-prevPoint);
263 
264  // Here we gould get the weight (=graph distance) from the PathFinderCommandSimplePath (if it were recording it), but the choice is to recompute it to keep more memory
265  float weight = offset==osg::Vec3() ? 0 : (*wg)(offset);
266 
267  // Math notes:
268  // speed = BASE_SPEED * offset.length() / weight;
269  // moveDist = speed * elapsed;
270  // movePointProgress = moveDist/offset.length();
271  //
272  // pointProgressSpeed = movePointProgress/elapsed;
273  // pointProgressSpeed = moveDist/offset.length()/elapsed;
274  // pointProgressSpeed = speed * elapsed/offset.length()/elapsed;
275  // pointProgressSpeed = speed/offset.length();
276  // pointProgressSpeed = BASE_SPEED / weight;
277 
278  const float prevPointProgress = pointProgress;
279  if (weight <=0) pointProgress = 1; // Set to next point
280  else pointProgress += BASE_SPEED / weight * elapsed;
281 
282  // "Consume" elapsed time
283  if (pointProgress <1) elapsed = 0;
284  else {
285  pointProgress -= 1;
286  elapsed -= (1 - prevPointProgress) / BASE_SPEED * weight;
287  ASSERT(elapsed >=0);
288  ++curPointInPath;
289  if (state == STOPPING || curPointInPath >= pathPoints.size()) {
290  pos = curPoint; // Set to final position
291  elapsed = 0;
292  doStop();
293  // Don't return, simply set the state == STOPPED
294  }
295  }
296 
297  if (state != STOPPED && elapsed<=0) pos = prevPoint + offset * pointProgress; // Only needed when elapsed==0 but we put it in the loop in order to avoid copying prevPoint/curPoint :)
298  }
299 
300  dGeomSetPosition(*geom, pos);
301  //osg::Vec3 oldPos = dGeomGetPositionV(*geom);
302  //static const float DELTA_POS_MIN2 = 5*5;
303  //static const float DELTA_POS_MAX = 20;
304  //osg::Vec3 offsetPos(pos-oldPos);
305  //float offsetPosLen2 = offsetPos.length2();
306  //if (offsetPosLen2 <= DELTA_POS_MIN2)
307  // dGeomSetPosition(*geom, pos);
308  //else {
309  // float offsetPosLen = sqrtf(offsetPosLen2);
310  // //if (offsetPosLen > DELTA_POS_MAX)
311  // static const float MAX_SPEED = BASE_SPEED * 1.5f;
312  // dGeomSetPosition(*geom, oldPos + offsetPos * osg::clampBelow(elapsed * MAX_SPEED / offsetPosLen, 1.f) );
313  //}
314 
315  // Update boid target
316  boidTarget->setPos(pos);
317  boidsUtil.targetUpdated(boidTarget);
318 
319  if (drawnPath) drawnPath->setMatrix(osg::Matrix::translate(-pos)); // Moved the path to the opposite so that it seems to be gloabal. A bit ugly but works :)
320 }
