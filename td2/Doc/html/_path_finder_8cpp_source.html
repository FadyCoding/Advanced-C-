25 void doSized(const osg::HeightField & hf, unsigned int col, unsigned int row, float radiusInCells, T & functor) {
26  ASSERT(radiusInCells>0);
27  ASSERT(hf.getRotation() == osg::Quat()); // Algorithm is simple and can't handle rotation yet
28  ASSERT(hf.getXInterval() == hf.getYInterval()); // Algorithm is simple and can't handle non-squares yet
29  float radius = radiusInCells*hf.getXInterval(); // In units
30 
31  // Determine min/max values of the for() loops
32  unsigned int startC = osg::clampAbove( static_cast<int>( floor(col - radiusInCells) ), 0 );
33  unsigned int startR = osg::clampAbove( static_cast<int>( floor(row - radiusInCells) ), 0 );
34  unsigned int endC = osg::clampBelow ( static_cast<int>( ceil (col + radiusInCells) ), static_cast<int>(hf.getNumColumns()-1) );
35  unsigned int endR = osg::clampBelow ( static_cast<int>( ceil (row + radiusInCells) ), static_cast<int>(hf.getNumRows() -1) );
36 
37  // Apply functor
38  osg::Vec3 center = hf.getVertex(col, row);
39  for(unsigned int c=startC; c<=endC; ++c) {
40  for(unsigned int r=startR; r<=endR; ++r) {
41  if ( (hf.getVertex(c, r) - center).length() <= radius ) functor(c, r);
42  }
43  }
44 }

49 bool doSquare(const osg::HeightField & hf, unsigned int col, unsigned int row, int radiusInCells, T & functor) {
50  ASSERT(radiusInCells>0);
51  ASSERT(hf.getRotation() == osg::Quat()); // Algorithm is simple and can't handle rotation yet
52  ASSERT(hf.getXInterval() == hf.getYInterval()); // Algorithm is simple and can't handle non-squares yet
53  //float radius = radiusInCells*hf.getXInterval(); // In units
54 
55  int theoricalLeft = static_cast<int>(col) - radiusInCells;
56  int theoricalUp = static_cast<int>(row) - radiusInCells;
57  int theoricalRight = static_cast<int>(col) + radiusInCells;
58  int theoricalDown = static_cast<int>(row) + radiusInCells;
59 
60  bool okLeft = theoricalLeft >= 0;
61  bool okUp = theoricalUp >= 0;
62  bool okRight = theoricalRight<= static_cast<int>(hf.getNumColumns()-1);
63  bool okDown = theoricalDown <= static_cast<int>(hf.getNumRows() -1);
64 
65  if (!(okLeft || okUp || okRight || okDown)) return false;
66 
67  // Determine min/max values of the for() loops
68  unsigned int startC = okLeft ? theoricalLeft : 0;
69  unsigned int startR = okUp ? theoricalUp : 0;
70  unsigned int endC = okRight ? theoricalRight: static_cast<int>(hf.getNumColumns()-1);
71  unsigned int endR = okDown ? theoricalDown : static_cast<int>(hf.getNumRows() -1);
72 
73  // Apply functor
74  if (okLeft) {
75  for(unsigned int r=startR; r<=endR; ++r) functor(startC, r);
76  }
77  if (okRight) {
78  for(unsigned int r=startR; r<=endR; ++r) functor(endC, r);
79  }
80  if (okUp) {
81  for(unsigned int c=startC+1; c<=endC-1; ++c) functor(c, startR);
82  }
83  if (okDown) {
84  for(unsigned int c=startC+1; c<=endC-1; ++c) functor(c, endR);
85  }
86  return true;
87 }

91 void doSurroundingVertices(const osg::HeightField & hf, unsigned int col, unsigned int row, T & functor) {
92  unsigned int maxR = hf.getNumRows()-1;
93  unsigned int maxC = hf.getNumColumns()-1;
94  bool okBorderN = row>0;
95  bool okBorderS = row<maxR;
96  bool okBorderE = col<maxC;
97  bool okBorderW = col>0;
98  if (okBorderN) functor(col, row, col , row-1); // North
99  if (okBorderE) {
100  if (okBorderN) functor(col, row, col+1, row-1); // N-E
101  /*No condition*/ functor(col, row, col+1, row ); // East
102  if (okBorderS) functor(col, row, col+1, row+1); // S-E
103  }
104  if (okBorderS) functor(col, row, col , row+1); // South
105  if (okBorderW) {
106  if (okBorderS) functor(col, row, col-1, row+1); // S-W
107  /*No condition*/ functor(col, row, col-1, row ); // West
108  if (okBorderN) functor(col, row, col-1, row-1); // N-W
109  }
110 }

164 void PathFinder::getClosestPointCoords(const osg::Vec3 & point, unsigned int & col, unsigned int & row) const {
165  // Note: this method used to be more compact, but I (Sukender) added multiple intermediate named variables to allow debugging :)
166  ASSERT(hf.getRotation() == osg::Quat()); // Algorithm is simple and can't handle rotation yet
167  auto nbCols = hf.getNumColumns();
168  auto nbRows = hf.getNumRows();
169  auto corner1 = hf.getVertex(0,0);
170  auto corner2 = hf.getVertex(nbCols-1,nbRows-1);
171  auto pointHFRef = point-corner1; // Point in HF coordinates, ie. using vertex(0,0) as origin.
172  auto delta = corner2-corner1;
173  ASSERT(delta.x() >0 && delta.y() >0);
174  auto cellSizeX = delta.x() / (nbCols-1);
175  auto cellSizeY = delta.y() / (nbRows-1);
176 
177  // HeightField algo makes x <=> col, y <=> row
178  col = osg::clampBetween<unsigned int>( static_cast<unsigned int>(round(pointHFRef.x() / cellSizeX)), 0, nbCols-1 );
179  row = osg::clampBetween<unsigned int>( static_cast<unsigned int>(round(pointHFRef.y() / cellSizeY)), 0, nbRows-1 );
180 }

192 void PathFinder::restoreDistanceUnidir(unsigned int col1, unsigned int row1, unsigned int col2, unsigned int row2) {
193  unsigned int v1 = getVertexDescriptor(col1, row1);
194  unsigned int v2 = getVertexDescriptor(col2, row2);
195  float distance = computeDistance(col1, row1, col2, row2);
196 
197  edge_descriptor e; bool exists;
198  tie(e, exists) = edge(v1, v2, g);
199  if (exists) {
200  // Edge exists: put the appropriate weight
201  if (distance >= 0) weightmap[e] = distance;
202  else weightmap[e] = FLT_MAX;
203  } else {
204  // Edge doesn't exists: create and fill it
205  if (distance >= 0) {
206  bool inserted;
207  tie(e, inserted) = add_edge(getVertexDescriptor(col1, row1), getVertexDescriptor(col2, row2), g);
208  ASSERT(inserted); // We did not use a set or map (but a list or vector)
209  weightmap[e] = distance;
210  }
