52 
53  if (elapsed >= .5) {
54  // FPS and mode display
55  std::string strGameMode;
56  unsigned int curFPS = static_cast<unsigned int>(FPS / elapsed);
57  if (curFPS >= app.getMaxFPS() * .95f && app.getMaxFPS()!=0) {
58 #ifdef _DEBUG
59  pText->setText(getAppName() + strGameMode + " - " + format_str(_("%1% FPS"), curFPS) + " (Max)" + subText, I18N::getEncoding());

64  pText->setText(getAppName() + strGameMode + " - " + format_str(_("%1% FPS"), curFPS) + subText, I18N::getEncoding());
