223  void addEdgeAndWeight(unsigned int col1, unsigned int row1, unsigned int col2, unsigned int row2) {
224  float distance = computeDistance(col1, row1, col2, row2);
225  if (distance >= 0) {
226  edge_descriptor e; bool inserted;
227  tie(e, inserted) = add_edge(getVertexDescriptor(col1, row1), getVertexDescriptor(col2, row2), g);
228  ASSERT(inserted); // We did not use a set or map (but a list or vector)
229  weightmap[e] = distance;
230  }
231  }
