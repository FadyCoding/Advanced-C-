65  if (isBuilt()) rechargeTime -= elapsed; // No need to clamp above 0
66  else {
67  ASSERT(_class);
68  // Compute elapsed build time
69  float buildElapsed(elapsed);
70  if ((_curBuildTime+buildElapsed) > _class->_buildTime) {
71  buildElapsed = _class->_buildTime - _curBuildTime;
72  }
73 
74  // Update current build time
75  _curBuildTime += elapsed;
76 
77  // Set life after the build time so that onLifeChanged() knows the building is finished
78  setLife(osg::clampBetween(getLife() + getMaxLife() * (buildElapsed / _class->_buildTime), 0.f, getMaxLife()));

80  // Update boid target (invert flee/attract multiplier when building)
81  ASSERT(boidTarget);
82  boidTarget->setMultiplier( fabs(boidTarget->getMultiplier()) * (isBuilt() ? -1 : 1) );
83  boidsUtil.targetUpdated(boidTarget);

86  //dGeomSetPosition(*geom, pos - osg::Z_AXIS * (length*.4f) + ???);
