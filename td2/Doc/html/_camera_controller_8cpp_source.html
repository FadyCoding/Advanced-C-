136 void CameraController::update(float elapsed, osg::Vec3 move, float scrollMove, bool viewMovedWithMouse, ERoll roll, EZoom zoom) {
137  float currentZoomLevel = eye.z() / BASE_HEIGHT;
138 
139  // Turn the translation according to dir (projected on Z=0)
140  osg::Vec3 projectedDir(dir.x(), dir.y(), 0);
141  projectedDir.normalize();
142  osg::Matrix horizRotation = osg::Matrix::rotate(osg::Y_AXIS, projectedDir);
143  move = move * horizRotation;
144 
145  if (viewMovedWithMouse) {
146  double prevSpeed = launchTrans.length();
147  double len = static_cast<double>(move.length());
148  //double newSpeed = osg::clampBelow(len, MAX_SPEED*currentZoomLevel);
149  double newSpeed = osg::clampBelow(len, MAX_SPEED*currentZoomLevel * FAST_MOVE_COEF);
150  if (newSpeed > prevSpeed*.33) launchTrans = move * (newSpeed / len);
151  else if (prevSpeed > 0) {
152  newSpeed = osg::clampAbove(prevSpeed*pow(SPEED_DECELERATION, static_cast<double>(elapsed)) - elapsed*DECELERATION*currentZoomLevel, 0.);
153  launchTrans *= newSpeed / prevSpeed;
154  }
155  } else {
156  if (move != osg::Vec3()) {
157  // User is moving the camera
158  //move.normalize();
159  double prevSpeed = launchTrans.length();
160  double newSpeed = osg::clampBelow(prevSpeed + elapsed*ACCELARATION*currentZoomLevel, MAX_SPEED*currentZoomLevel);
161  launchTrans = move * newSpeed;
162  // TODO avoid accelerating when going the opposite direction of the current one
163  } else {
164  double prevSpeed = launchTrans.length();
165  if (prevSpeed > 0) {
166  double newSpeed = osg::clampAbove(prevSpeed*pow(SPEED_DECELERATION, static_cast<double>(elapsed)) - elapsed*DECELERATION*currentZoomLevel, 0.);
167  launchTrans *= newSpeed / prevSpeed;
168  }
169  }
170  }
171  eye += launchTrans;
172 
173  eye.x() = osg::clampBetween(eye.x(), static_cast<osg::Vec3::value_type>(bounds.xMin()), static_cast<osg::Vec3::value_type>(bounds.xMax()));
174  eye.y() = osg::clampBetween(eye.y(), static_cast<osg::Vec3::value_type>(bounds.yMin()), static_cast<osg::Vec3::value_type>(bounds.yMax()));
175 
176  // Rotation
177  if (roll == ROLL_LEFT) {
178  launchRot = osg::clampBelow(launchRot + elapsed * ROT_ACCELARATION, ROT_MAX_SPEED);
179  }
180  else if (roll == ROLL_RIGHT) {
181  launchRot = osg::clampAbove(launchRot - elapsed * ROT_ACCELARATION, -ROT_MAX_SPEED);
182  }
183  else {
184  if (launchRot >= 0) {
185  launchRot = osg::clampAbove(launchRot - elapsed * ROT_DECELERATION, 0.);
186  } else {
187  launchRot = osg::clampBelow(launchRot + elapsed * ROT_DECELERATION, 0.);
188  }
189  }
190  dir = dir * osg::Matrix::rotate(launchRot, osg::Z_AXIS);
191 
192  // Zoom
193  if (zoom == ZOOM_IN) {
194  launchZoom = osg::clampBelow(launchZoom - elapsed * ZOOM_ACCELARATION, -ZOOM_MAX_SPEED);
195  }
196  else if (zoom == ZOOM_OUT) {
197  launchZoom = osg::clampAbove(launchZoom + elapsed * ZOOM_ACCELARATION, ZOOM_MAX_SPEED);
198  }
199  else if (scrollMove != 0) {
200  launchZoom = osg::clampBetween(launchZoom + scrollMove * 20 * elapsed * ZOOM_ACCELARATION, -ZOOM_MAX_SPEED, ZOOM_MAX_SPEED);
201  }
202  else {
203  if (launchZoom >= 0) {
204  launchZoom = osg::clampAbove(launchZoom - elapsed * ZOOM_DECELERATION, 0.);
205  } else {
206  launchZoom = osg::clampBelow(launchZoom + elapsed * ZOOM_DECELERATION, 0.);
207  }
208  }
209  eye.z() = osg::clampBetween(eye.z() + launchZoom, MIN_ZOOM, MAX_ZOOM);
210  //currentZoomLevel = eye.z() / BASE_HEIGHT;
211 
212  // Tilt the camera according to currentZoomLevel
213  //dir = dir * osg::Matrix::rotate(1/currentZoomLevel * elapsed * .01f, osg::X_AXIS * horizRotation);
214  //float targetAngle = 1/currentZoomLevel;
215  //dir = osg::Vec3(dir.x(), dir.y(), 0) * osg::Matrix::rotate(1/currentZoomLevel, osg::X_AXIS * horizRotation);
216 
217  dir.z() = -osg::clampBetween(sigmoid(currentZoomLevel*2.f-.2f), 0.2f, .98f);
218  dir.normalize();
219 
220  currentZoomLevel = eye.z() / BASE_HEIGHT;
221 }
